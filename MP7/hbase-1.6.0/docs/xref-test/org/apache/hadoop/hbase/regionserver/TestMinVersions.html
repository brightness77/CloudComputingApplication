69      HTableDescriptor htd =
70          hbu.createTableDescriptor(name.getMethodName(), 1, 1000, 1, KeepDeletedCells.FALSE);
71      HRegion region = hbu.createLocalHRegion(htd, null, null);
72      try {
73  
74        // 2s in the past
75        long ts = EnvironmentEdgeManager.currentTime() - 2000;

77        Put p = new Put(T1, ts);
78        p.add(c0, c0, T1);
79        region.put(p);
80  
81        p = new Put(T1, ts+1);
82        p.add(c0, c0, T4);
83        region.put(p);
84  
85        p = new Put(T3, ts);
86        p.add(c0, c0, T3);
87        region.put(p);
88  
89        // now make sure that getClosestBefore(...) get can
90        // rows that would be expired without minVersion.
91        // also make sure it gets the latest version
92        Result r = region.getClosestRowBefore(T1, c0);
93        checkResult(r, c0, T4);
94  
95        r = region.getClosestRowBefore(T2, c0);
96        checkResult(r, c0, T4);
97  
98        // now flush/compact
99        region.flush(true);
100       region.compact(true);
101 
102       r = region.getClosestRowBefore(T1, c0);
103       checkResult(r, c0, T4);
104 
105       r = region.getClosestRowBefore(T2, c0);

306   @Test
307   public void testBaseCase() throws Exception {
308     // 1 version minimum, 1000 versions maximum, ttl = 1s
309     HTableDescriptor htd =
310         hbu.createTableDescriptor(name.getMethodName(), 2, 1000, 1, KeepDeletedCells.FALSE);
311     HRegion region = hbu.createLocalHRegion(htd, null, null);
312     try {
313 
314       // 2s in the past
315       long ts = EnvironmentEdgeManager.currentTime() - 2000;
316 
317        // 1st version
318       Put p = new Put(T1, ts-3);
319       p.add(c0, c0, T1);
320       region.put(p);
321 
322       // 2nd version
323       p = new Put(T1, ts-2);
324       p.add(c0, c0, T2);
325       region.put(p);
326 
327       // 3rd version
328       p = new Put(T1, ts-1);
329       p.add(c0, c0, T3);
330       region.put(p);
331 
332       // 4th version
333       p = new Put(T1, ts);
334       p.add(c0, c0, T4);
335       region.put(p);
336 
337       Result r = region.get(new Get(T1));
338       checkResult(r, c0, T4);
339 
340       Get g = new Get(T1);
341       g.setTimeRange(0L, ts+1);
342       r = region.get(g);
343       checkResult(r, c0, T4);
344 
345   // oldest version still exists
346       g.setTimeRange(0L, ts-2);
347       r = region.get(g);
348       checkResult(r, c0, T1);
349 
350       // gets see only available versions
351       // even before compactions
352       g = new Get(T1);
353       g.setMaxVersions();
354       r = region.get(g); // this'll use ScanWildcardColumnTracker
355       checkResult(r, c0, T4,T3);
356 
357       g = new Get(T1);
358       g.setMaxVersions();
359       g.addColumn(c0, c0);
360       r = region.get(g);  // this'll use ExplicitColumnTracker
361       checkResult(r, c0, T4,T3);
362 
363       // now flush
364       region.flush(true);
365 
366       // with HBASE-4241 a flush will eliminate the expired rows
367       g = new Get(T1);
368       g.setTimeRange(0L, ts-2);
369       r = region.get(g);
370       assertTrue(r.isEmpty());
371 
372       // major compaction
373       region.compact(true);
374 
375       // after compaction the 4th version is still available
376       g = new Get(T1);
377       g.setTimeRange(0L, ts+1);
378       r = region.get(g);
379       checkResult(r, c0, T4);
380 
381       // so is the 3rd
382       g.setTimeRange(0L, ts);
383       r = region.get(g);
384       checkResult(r, c0, T3);
385 
386       // but the 2nd and earlier versions are gone
387       g.setTimeRange(0L, ts-1);
388       r = region.get(g);
389       assertTrue(r.isEmpty());
390     } finally {
391       HRegion.closeHRegion(region);
392     }
393   }

470   @Test
471   public void testMinVersionsWithKeepDeletedCellsTTL() throws Exception {
472     int ttl = 4;
473     HTableDescriptor htd = hbu.createTableDescriptor(name.getMethodName(),
474       2, Integer.MAX_VALUE, ttl, KeepDeletedCells.TTL);
475 
476     HRegion region = hbu.createLocalHRegion(htd, null, null);
477 
478     long startTS = EnvironmentEdgeManager.currentTime();
479     ManualEnvironmentEdge injectEdge = new ManualEnvironmentEdge();
480     injectEdge.setValue(startTS);
481     EnvironmentEdgeManager.injectEdge(injectEdge);
482 
483     long ts = startTS - 2000;
484     // 1st version
485     Put p = new Put(T1, ts-3);
486     p.addColumn(c0, c0, T1);
487     region.put(p);
488 
489     // 2nd version
490     p = new Put(T1, ts-2);
491     p.addColumn(c0, c0, T2);
492     region.put(p);
493 
494     // 3rd version
495     p = new Put(T1, ts-1);
496     p.addColumn(c0, c0, T3);
497     region.put(p);
498 
499     Get g;
500     Result r;
501 
502     //check we can still see all versions before compaction
503     g = new Get(T1);
504     g.setMaxVersions();
505     g.setTimeRange(0, ts);
506     r = region.get(g);
507     checkResult(r, c0, T3, T2, T1);
508 
509     region.flush(true);
510     region.compact(true);
511     assertEquals(startTS, EnvironmentEdgeManager.currentTime());
512     long expiredTime = EnvironmentEdgeManager.currentTime() - ts - 3;
513     assertTrue("TTL for T1 has expired", expiredTime < (ttl * 1000));
514     //check that nothing was purged yet
515     g = new Get(T1);
516     g.setMaxVersions();
517     g.setTimeRange(0, ts);
518     r = region.get(g);
519     checkResult(r, c0, T3, T2, T1);
520 
521     g = new Get(T1);
522     g.setMaxVersions();
523     g.setTimeRange(0, ts -1);
524     r = region.get(g);
525     checkResult(r, c0, T2, T1);
526 
527     injectEdge.incValue(ttl * 1000);
528 
529     region.flush(true);
530     region.compact(true);
531 
532     //check that after compaction (which is after TTL) that only T1 was purged
533     g = new Get(T1);
534     g.setMaxVersions();
535     g.setTimeRange(0, ts);
536     r = region.get(g);
537     checkResult(r, c0, T3, T2);
538 
539     g = new Get(T1);
540     g.setMaxVersions();
541     g.setTimeStamp(ts -2);
542     r = region.get(g);
543     checkResult(r, c0, T2);
544   }

547   private void checkResult(Result r, byte[] col, byte[] ... vals) {
548     assertEquals(r.size(), vals.length);
549     List<Cell> kvs = r.getColumnCells(col, col);
550     assertEquals(kvs.size(), vals.length);
551     for (int i=0;i<vals.length;i++) {
552       String expected = Bytes.toString(vals[i]);
553       String actual = Bytes.toString(CellUtil.cloneValue(kvs.get(i)));
554       assertTrue(expected + " was expected but doesn't match " + actual,
555         CellUtil.matchingValue(kvs.get(i), vals[i]));
556     }
557   }
